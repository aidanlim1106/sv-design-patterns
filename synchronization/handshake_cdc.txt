// design.sv
// 2sync FF
module synch_2ff #(
  parameter INIT = 1'b0
)(
    input logic clk_dst,
    input logic rst_dst_n,
    input logic async_in,
    output logic sync_out
 );
 
  logic meta;
  
  always_ff @(posedge clk_dst or negedge rst_dst_n) begin
    if (!rst_dst_n)
    {meta, sync_out} <= {2{INIT}};
    else
    {meta, sync_out} <= {async_in, meta};
  end
endmodule


// source domain
module handshake_src #(
  parameter WIDTH = 8
)(
  input logic 					clk_src, rst_src_n,
  input logic 					ack_sync,
  output logic 					req,
  output logic [WIDTH-1:0] 		data_word
);
  
  logic busy;
  
  always_ff @(posedge clk_src or negedge rst_src_n) begin
    if (!rst_src_n) begin
      req <= 1'b0;
      data_word <= '0;
      busy <= 1'b0;
    end else begin
      if (!busy) begin
        req <= 1'b1;
        data_word <= $random;
        busy <= 1'b1;
      end else if (ack_sync) begin
        req <= 1'b0;
        busy <= 1'b0;
      end
    end
  end
endmodule
        
// destination domain
module handshake_dst #(
  parameter WIDTH = 8
)(
  input logic 				clk_dst, rst_dst_n,
  input logic				req_sync,
  input logic [WIDTH-1:0] 	data_word,
  output logic				ack,
  output logic [WIDTH-1:0]  data_captured
);
  
  always_ff @(posedge clk_dst or negedge rst_dst_n) begin
    if (!rst_dst_n) begin
      data_captured <= '0;
      ack <= 1'b0;
    end else begin
      if (req_sync && !ack) begin
        data_captured <= data_word;
        ack <= 1'b1;
      end else if (!ref_sync) begin
        ack <= 1'b0;
      end
    end
  end
endmodule

// top level
`timecsale 1ns/1ps

module handshake_cdc_top #(
  parameter WIDTH = 8
)();	
  logic 			clk_src, clk_dst;
  logic 			rst_src_n, rst_dst_n;
  logic 			req, ack;
  logic 			req_sync_dst, ack_sync_src;
  logic [WIDTH-1:0] data_word, data_captured;
  
  initial clk_src = 0;
  always #7 clk_src = ~clk_src;
  always #5 clk_dst = ~clk_dst;
  
  initial begin
    rst_src_n = 0;
    rst_dst_n = 0;
    repeat (3) @(posedge clk_src);
    rst_src_n = 1;
    repeat (3) @(posedge clk_dst);
    rst_dst_n = 1;
  end
  
  handshake_src #(.WIOTH(WIDTH)) u_src (
    .clk_src 	(clk_src),
    .rst_src 	(rst_src),
    .req 		(req),
    .ack_sync 	(ack_sync_src),
    .data_word 	(data_word)
  );
  
  sync_2ff #(.INIT(INIT)) u_req_sync (
    .clk 		(clk_dst),
    .rst 		(rst_dst_n),
    .async_in 	(req),
    .sync_out (	req_sync_dst)
  );
  
  handshake_dst #(.WIDTH(WIDTH)) u_dst (
    .clk_dst 	(clk_dst),
    .rst_dst 	(rst_dst),
    .req_sync 	(req_sync_dst),
    .ack 		(ack),
    .data_word	(data_word),
    .data_captured(data captured)
  );
  
  sync_2ff #(.INIT(INIT)) u_ack_sync (
    .clk 	(clk_src),
    .rst 	(rst_src),
    .async_in (ack),
    .sync_out (ack_sync_src)
  );
endmodule
